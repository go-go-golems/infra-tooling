     1	#!/usr/bin/env bash
     2	set -euo pipefail
     3	
     4	# Extract Codex-authored review bodies and inline code-review comments for one or
     5	# more PRs. This is the evidence-gathering script used during the XGOJA-015
     6	# release train when the readiness checker reported CODEX_FEEDBACK.
     7	#
     8	# Usage:
     9	#   08-extract-codex-review-comments.sh <pr-url-or-owner/repo#number> [...]
    10	#   08-extract-codex-review-comments.sh --file /tmp/xgoja-release-prs.txt
    11	
    12	CODEX_RE='codex|chatgpt|openai'
    13	
    14	parse_pr() {
    15	  local pr="$1"
    16	  if [[ "$pr" =~ github\.com/([^/]+)/([^/]+)/pull/([0-9]+) ]]; then
    17	    printf '%s %s %s\n' "${BASH_REMATCH[1]}" "${BASH_REMATCH[2]}" "${BASH_REMATCH[3]}"
    18	    return 0
    19	  fi
    20	  if [[ "$pr" =~ ^([^/]+)/([^#]+)#([0-9]+)$ ]]; then
    21	    printf '%s %s %s\n' "${BASH_REMATCH[1]}" "${BASH_REMATCH[2]}" "${BASH_REMATCH[3]}"
    22	    return 0
    23	  fi
    24	  echo "could not parse PR: $pr" >&2
    25	  return 2
    26	}
    27	
    28	read_prs_file() {
    29	  local file="$1"
    30	  while IFS= read -r pr || [[ -n "$pr" ]]; do
    31	    pr="${pr%%#*}"
    32	    pr="$(printf '%s' "$pr" | xargs)"
    33	    [[ -z "$pr" ]] && continue
    34	    printf '%s\n' "$pr"
    35	  done < "$file"
    36	}
    37	
    38	print_pr() {
    39	  local pr="$1" owner repo number
    40	  read -r owner repo number < <(parse_pr "$pr")
    41	  echo "### $owner/$repo#$number"
    42	  gh api graphql \
    43	    -f owner="$owner" \
    44	    -f repo="$repo" \
    45	    -F number="$number" \
    46	    -f query='query($owner:String!,$repo:String!,$number:Int!){ repository(owner:$owner,name:$repo){ pullRequest(number:$number){ headRefOid reviews(last:20){nodes{author{login} state body url submittedAt comments(first:100){nodes{path line body url author{login}}}}}}}}' |
    47	    jq -r --arg re "$CODEX_RE" '
    48	      .data.repository.pullRequest as $pr |
    49	      $pr.reviews.nodes[] |
    50	      select((.author.login // "") | test($re; "i")) |
    51	      "REVIEW " + .submittedAt + " " + .url,
    52	      "BODY: " + ((.body // "") | gsub("\\n"; " ") | .[0:500]),
    53	      "COMMENTS:",
    54	      ((.comments.nodes // [])[]? | "- " + (.path // "<unknown>") + ":" + ((.line // "?")|tostring) + " " + ((.body // "") | gsub("\\n"; " ") | .[0:700]) + " " + (.url // "")),
    55	      "---"
    56	    '
    57	}
    58	
    59	if [[ $# -eq 0 ]]; then
    60	  echo "usage: $0 <pr> [...] | --file <prs-file>" >&2
    61	  exit 2
    62	fi
    63	
    64	if [[ "${1:-}" == "--file" ]]; then
    65	  [[ $# -eq 2 ]] || { echo "--file requires exactly one argument" >&2; exit 2; }
    66	  while IFS= read -r pr; do
    67	    print_pr "$pr"
    68	  done < <(read_prs_file "$2")
    69	else
    70	  for pr in "$@"; do
    71	    print_pr "$pr"
    72	  done
    73	fi

    27	  return 2
